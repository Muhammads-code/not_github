        // TO IGNORE FILE :-
        [1] touch .gitignore
        [2] in .gitignore file type name.html

        // TO IGNORE MULTIPLE FILES :-
        [3] in .gitignore file type *.html

        // TO IGNORE FOLDERS :-
        [4] in .gitignore file type /folder name

        // TO MAKE BRANCH :-
        [5] git branch master1 = (creates master1)

        // TO SWITCH BRANCHS :-
        [6] git checkout master1 = (takes master to master1)
